00094 int
00095 netsnmp_table_row_register(netsnmp_handler_registration *reginfo,
00096                            netsnmp_table_registration_info *tabreg,
00097                            void *row, netsnmp_variable_list *index)
00098 {
00099     netsnmp_handler_registration *reg2;
00100     netsnmp_mib_handler *handler;
00101     oid    row_oid[MAX_OID_LEN];
00102     size_t row_oid_len, len;
00103     char   tmp[SNMP_MAXBUF_MEDIUM];
00104 
00105     if ((NULL == reginfo) || (NULL == reginfo->handler) || (NULL == tabreg)) {
00106         snmp_log(LOG_ERR, "bad param in netsnmp_table_row_register\n");
00107         return SNMPERR_GENERR;
00108     }
00109 
00110         /*
00111          *   The first table_row invoked for a particular table should
00112          * register the full table as well, with a default handler to
00113          * process requests for non-existent (or incomplete) rows.
00114          *
00115          *   Subsequent table_row registrations attempting to set up
00116          * this default handler would fail - preferably silently!
00117          */
00118     snprintf(tmp, sizeof(tmp), "%s_table", reginfo->handlerName);
00119     reg2 = netsnmp_create_handler_registration(
00120               tmp,     _table_row_default_handler,
00121               reginfo->rootoid, reginfo->rootoid_len,
00122               reginfo->modes);
00123     netsnmp_register_table(reg2, tabreg);  /* Ignore return value */
00124 
00125         /*
00126          * Adjust the OID being registered, to take account
00127          * of the indexes and column range provided....
00128          */
00129     row_oid_len = reginfo->rootoid_len;
00130     memcpy( row_oid, (u_char *) reginfo->rootoid, row_oid_len * sizeof(oid));
00131     row_oid[row_oid_len++] = 1;   /* tableEntry */
00132     row_oid[row_oid_len++] = tabreg->min_column;
00133     reginfo->range_ubound  = tabreg->max_column;
00134     reginfo->range_subid   = row_oid_len-1;
00135     build_oid_noalloc(&row_oid[row_oid_len],
00136                       MAX_OID_LEN-row_oid_len, &len, NULL, 0, index);
00137     row_oid_len += len;
00138     free(reginfo->rootoid);
00139     reginfo->rootoid = snmp_duplicate_objid(row_oid, row_oid_len);
00140     reginfo->rootoid_len = row_oid_len;
00141 
00142      
00143         /*
00144          * ... insert a minimal handler ...
00145          */
00146     handler = netsnmp_table_row_handler_get(row);
00147     netsnmp_inject_handler(reginfo, handler );
00148 
00149         /*
00150          * ... and register the row
00151          */
00152     return netsnmp_register_handler(reginfo);
00153 }

00174 static int
00175 _table_row_handler(netsnmp_mib_handler          *handler,
00176                    netsnmp_handler_registration *reginfo,
00177                    netsnmp_agent_request_info   *reqinfo,
00178                    netsnmp_request_info         *requests)
00179 {
00180     int             rc = SNMP_ERR_NOERROR;
00181     netsnmp_request_info *req;
00182     void                 *row;
00183 
00185     netsnmp_assert((NULL != handler) && (NULL != handler->myvoid));
00186     netsnmp_assert((NULL != reginfo) && (NULL != reqinfo));
00187 
00188     DEBUGMSGTL(("table_row", "Mode %s, Got request:\n",
00189                 se_find_label_in_slist("agent_mode",reqinfo->mode)));
00190 
00191     /*
00192      * First off, get our pointer from the handler.
00193      * This contains the row that was actually registered.
00194      * Make this available for each of the requests passed in.
00195      */
00196     row = handler->myvoid;
00197     for (req = requests; req; req=req->next)
00198         netsnmp_request_add_list_data(req,
00199                 netsnmp_create_data_list(TABLE_ROW_DATA, row, NULL));
00200 
00201     /*
00202      * Then call the next handler, to actually process the request
00203      */
00204     rc = netsnmp_call_next_handler(handler, reginfo, reqinfo, requests);
00205     if (rc != SNMP_ERR_NOERROR) {
00206         DEBUGMSGTL(("table_row", "next handler returned %d\n", rc));
00207     }
00208 
00209     return rc;
00210 }

00213 _table_row_default_handler(netsnmp_mib_handler  *handler,

00215                    netsnmp_agent_request_info   *reqinfo,

00218     netsnmp_request_info       *req;

00220     netsnmp_table_registration_info *tabreg;
00221 
00222     tabreg = netsnmp_find_table_registration_info(reginfo);
00223     for ( req=requests; req; req=req->next ) {
00224         table_info = netsnmp_extract_table_info( req );
00225         if (( table_info->colnum >= tabreg->min_column ) ||
00226             ( table_info->colnum <= tabreg->max_column )) {
00227             netsnmp_set_request_error( reqinfo, req, SNMP_NOSUCHINSTANCE );
