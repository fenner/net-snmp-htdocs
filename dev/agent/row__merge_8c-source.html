00057 int
00058 netsnmp_row_merge_helper_handler(netsnmp_mib_handler *handler,
00059                                  netsnmp_handler_registration *reginfo,
00060                                  netsnmp_agent_request_info *reqinfo,
00061                                  netsnmp_request_info *requests)
00062 {
00063     netsnmp_request_info *request, **saved_requests;
00064     char *saved_status;
00065     int i, j, ret, tail, count = 0;
00066 
00067     /*
00068      * xxx-rks - for sets, should store this info in agent request info, so it
00069      *           doesn't need to be done for every mode.
00070      *
00071      * XXX - Need some mechanism for specifying the length of the table OID
00072      *  i.e. where to start looking for shared indexes
00073      *  N.B: The first subidentifier after the table OID will typically vary,
00074      *    being the column subidentifier, so skip this as well.
00075      */
00076     /* int SKIP_OID = reginfo->rootoid_len + 2; */
00077     int SKIP_OID = (int)handler->myvoid + 1;
00078 
00079     DEBUGMSGTL(("helper:row_merge", "Got request (%d)\n", SKIP_OID));
00080     DEBUGMSGOID(("helper:row_merge", reginfo->rootoid, reginfo->rootoid_len));
00081     DEBUGMSG(("helper:row_merge", "\n"));
00082 
00083     /*
00084      * Count the requests, and set up an array to keep
00085      *  track of the original order.
00086      */
00087     for (request = requests; request; request = request->next) 
00088         count++;
00089 
00090     /*
00091      * Optimization: skip all this if there is just one request
00092      */
00093     if(count == 1) {
00094         DEBUGMSGTL(("helper:row_merge", "  only one varbind\n"));
00095         return netsnmp_call_next_handler(handler, reginfo, reqinfo, requests);
00096     }
00097 
00098     /*
00099      * allocate memory for saved structure
00100      */
00101     saved_requests = (netsnmp_request_info**)calloc(count+1, sizeof(netsnmp_request_info*));
00102     saved_status   =                  (char*)calloc(count,   sizeof(char));
00103 
00104     /*
00105      * set up saved requests, and set any processed requests to done
00106      */
00107     i = 0;
00108     for (request = requests; request; request = request->next, i++) {
00109         if (request->processed) {
00110             saved_status[i] = ROW_MERGE_DONE;
00111             DEBUGMSGTL(("helper:row_merge", "  skipping processed oid: "));
00112             DEBUGMSGOID(("helper:row_merge", request->requestvb->name,
00113                          request->requestvb->name_length));
00114             DEBUGMSG(("helper:row_merge", "\n"));
00115         }
00116         saved_requests[i] = request;
00117     }
00118 
00119     /*
00120      * Note that saved_requests[count] is valid
00121      *    (because of the 'count+1' in the calloc above),
00122      * but NULL (since it's past the end of the list).
00123      * This simplifies the re-linking later.
00124      */
00125 
00126     /*
00127      * Work through the (unprocessed) requests in order.
00128      * For each of these, search the rest of the list for any
00129      *   matching indexes, and link them into a new list.
00130      */
00131     for (i=0; i<count; i++) {
00132         if (saved_status[i] != ROW_MERGE_WAITING) {
00133             /*
00134              * Already processed, so just re-link into the original list
00135              */
00136             saved_requests[i]->next = saved_requests[i+1];
00137             continue;
00138         }
00139 
00140         DEBUGMSGTL(("helper:row_merge", "  oid[%d]: ", i));
00141         DEBUGMSGOID(("helper:row_merge", saved_requests[i]->requestvb->name, saved_requests[i]->requestvb->name_length));
00142         DEBUGMSG(("helper:row_merge", "\n"));
00143 
00144         saved_requests[i]->next = NULL;
00145         saved_status[i] = ROW_MERGE_ACTIVE;
00146         tail = i;
00147         for (j=i+1; j<count; j++) {
00148             if (saved_status[j] != ROW_MERGE_WAITING) {
00149                 continue;
00150             }
00151             DEBUGMSGTL(("helper:row_merge", "? oid[%d]: ", j));
00152             DEBUGMSGOID(("helper:row_merge", saved_requests[j]->requestvb->name, saved_requests[j]->requestvb->name_length));
00153             DEBUGMSG(("helper:row_merge", "\n"));
00154             if (!snmp_oid_compare(
00155                     saved_requests[i]->requestvb->name+SKIP_OID,
00156                     saved_requests[i]->requestvb->name_length-SKIP_OID,
00157                     saved_requests[j]->requestvb->name+SKIP_OID,
00158                     saved_requests[j]->requestvb->name_length-SKIP_OID)) {
00159                 DEBUGMSGTL(("helper:row_merge", "merged\n"));
00160                 saved_requests[tail]->next = saved_requests[j];
00161                 saved_requests[j]->next    = NULL;
00162                 saved_status[j] = ROW_MERGE_ACTIVE;
00163                 tail = j;
00164             }
00165         }
00166 
00167         /*
00168          * call the next handler with this list, and 
00169          * restore the original next pointer 
00170          */
00171         ret = netsnmp_call_next_handler(handler, reginfo, reqinfo,
00172                                         saved_requests[i]);
00173         saved_requests[i]->next = saved_requests[i+1];
00174 
00175         if (ret != SNMP_ERR_NOERROR) {
00176             /* 
00177              * Something went wrong.
00178              * Re-link the rest of the original list,
00179              *   clean up, and report back.
00180              */
00181             for (j=0; j<count; j++)
00182                 saved_requests[j]->next = saved_requests[j+1];
00183             free(saved_requests);
00184             free(saved_status);
00185             return ret;
00186         }
00187     }
00188 
00189     free(saved_requests);
00190     free(saved_status);
00191     return SNMP_ERR_NOERROR;
00192 }
