00101     /*
00102      * int snmp_select_info(numfds, fdset, timeout, block)
00103      * int *numfds;
00104      * fd_set   *fdset;
00105      * struct timeval *timeout;
00106      * int *block;
00107      *
00108      * Returns info about what snmp requires from a select statement.
00109      * numfds is the number of fds in the list that are significant.
00110      * All file descriptors opened for SNMP are OR'd into the fdset.
00111      * If activity occurs on any of these file descriptors, snmp_read
00112      * should be called with that file descriptor set.
00113      *
00114      * The timeout is the latest time that SNMP can wait for a timeout.  The
00115      * select should be done with the minimum time between timeout and any other
00116      * timeouts necessary.  This should be checked upon each invocation of select.
00117      * If a timeout is received, snmp_timeout should be called to check if the
00118      * timeout was for SNMP.  (snmp_timeout is idempotent)
00119      *
00120      * Block is 1 if the select is requested to block indefinitely, rather than
00121      * time out.  If block is input as 1, the timeout value will be treated as
00122      * undefined, but it must be available for setting in snmp_select_info.  On
00123      * return, if block is true, the value of timeout will be undefined.
00124      *
00125      * snmp_select_info returns the number of open sockets.  (i.e. The number
00126      * of sessions open)
