00038 long
00039 auto_nlist_value(const char *string)
00040 {
00041     struct autonlist **ptr, *it = 0;
00042     int             cmp;
00043 
00044     if (string == 0)
00045         return 0;
00046 
00047     ptr = &nlists;
00048     while (*ptr != 0 && it == 0) {
00049         cmp = strcmp((*ptr)->symbol, string);
00050         if (cmp == 0)
00051             it = *ptr;
00052         else if (cmp < 0) {
00053             ptr = &((*ptr)->left);
00054         } else {
00055             ptr = &((*ptr)->right);
00056         }
00057     }
00058     if (*ptr == 0) {
00059         *ptr = (struct autonlist *) malloc(sizeof(struct autonlist));
00060         it = *ptr;
00061         it->left = 0;
00062         it->right = 0;
00063         it->symbol = (char *) malloc(strlen(string) + 1);
00064         strcpy(it->symbol, string);
00065         /*
00066          * allocate an extra byte for inclusion of a preceding '_' later 
00067          */
00068         it->nl[0].n_name = (char *) malloc(strlen(string) + 2);
00069 #ifdef aix4
00070         strcpy(it->nl[0].n_name, string);
00071 #else
00072         sprintf(it->nl[0].n_name, "_%s", string);
00073 #endif
00074         it->nl[1].n_name = 0;
00075         init_nlist(it->nl);
00076 #ifndef aix4
00077         if (it->nl[0].n_type == 0) {
00078             strcpy(it->nl[0].n_name, string);
00079             init_nlist(it->nl);
00080         }
00081 #endif
00082         if (it->nl[0].n_type == 0) {
00083             if (!netsnmp_ds_get_boolean(NETSNMP_DS_APPLICATION_ID, 
00084                                         NETSNMP_DS_AGENT_NO_ROOT_ACCESS)) {
00085                 snmp_log(LOG_ERR, "nlist err: neither %s nor _%s found.\n",
00086                          string, string);
00087             }
00088             return (-1);
00089         } else {
00090             DEBUGMSGTL(("auto_nlist", "nlist:  found symbol %s at %x.\n",
00091                         it->symbol, it->nl[0].n_value));
00092             return (it->nl[0].n_value);
00093         }
00094     } else
00095         return (it->nl[0].n_value);
00096 }
