00062  * generate_Ku
00063  *
00064  * Parameters:
00065  *      *hashtype       MIB OID for the transform type for hashing.
00066  *       hashtype_len   Length of OID value.
00067  *      *P              Pre-allocated bytes of passpharase.
00068  *       pplen          Length of passphrase.
00069  *      *Ku             Buffer to contain Ku.
00070  *      *kulen          Length of Ku buffer.
00071  *      
00072  * Returns:
00073  *      SNMPERR_SUCCESS                 Success.
00074  *      SNMPERR_GENERR                  All errors.
00075  *
00076  *
00077  * Convert a passphrase into a master user key, Ku, according to the
00078  * algorithm given in RFC 2274 concerning the SNMPv3 User Security Model (USM)
00079  * as follows:
00080  *
00081  * Expand the passphrase to fill the passphrase buffer space, if necessary,
00082  * concatenation as many duplicates as possible of P to itself.  If P is
00083  * larger than the buffer space, truncate it to fit.
00084  *
00085  * Then hash the result with the given hashtype transform.  Return
00086  * the result as Ku.
00087  *
00088  * If successful, kulen contains the size of the hash written to Ku.
00089  *
00090  * NOTE  Passphrases less than USM_LENGTH_P_MIN characters in length
00091  *       cause an error to be returned.
00092  *       (Punt this check to the cmdline apps?  XXX)

00241 /*******************************************************************-o-******
00242  * generate_kul
00243  *
00244  * Parameters:
00245  *      *hashtype
00246  *       hashtype_len
00247  *      *engineID
00248  *       engineID_len
00249  *      *Ku             Master key for a given user.
00250  *       ku_len         Length of Ku in bytes.
00251  *      *Kul            Localized key for a given user at engineID.
00252  *      *kul_len        Length of Kul buffer (IN); Length of Kul key (OUT).
00253  *      
00254  * Returns:
00255  *      SNMPERR_SUCCESS                 Success.
00256  *      SNMPERR_GENERR                  All errors.
00257  *
00258  *
00259  * Ku MUST be the proper length (currently fixed) for the given hashtype.
00260  *
00261  * Upon successful return, Kul contains the localized form of Ku at
00262  * engineID, and the length of the key is stored in kul_len.
00263  *
00264  * The localized key method is defined in RFC2274, Sections 2.6 and A.2, and
00265  * originally documented in:
00266  *      U. Blumenthal, N. C. Hien, B. Wijnen,
00267  *      "Key Derivation for Network Management Applications",
00268  *      IEEE Network Magazine, April/May issue, 1997.
00269  *
00270  *
00271  * ASSUMES  SNMP_MAXBUF >= sizeof(Ku + engineID + Ku).
00272  *
00273  * NOTE  Localized keys for privacy transforms are generated via
00274  *       the authentication transform held by the same usmUser.
00275  *
00276  * XXX  An engineID of any length is accepted, even if larger than
00277  *      what is spec'ed for the textual convention.
00278  */

00349 /*******************************************************************-o-******
00350  * encode_keychange
00351  *
00352  * Parameters:
00353  *      *hashtype       MIB OID for the hash transform type.
00354  *       hashtype_len   Length of the MIB OID hash transform type.
00355  *      *oldkey         Old key that is used to encodes the new key.
00356  *       oldkey_len     Length of oldkey in bytes.
00357  *      *newkey         New key that is encoded using the old key.
00358  *       newkey_len     Length of new key in bytes.
00359  *      *kcstring       Buffer to contain the KeyChange TC string.
00360  *      *kcstring_len   Length of kcstring buffer.
00361  *      
00362  * Returns:
00363  *      SNMPERR_SUCCESS                 Success.
00364  *      SNMPERR_GENERR                  All errors.
00365  *
00366  *
00367  * Uses oldkey and acquired random bytes to encode newkey into kcstring
00368  * according to the rules of the KeyChange TC described in RFC 2274, Section 5.
00369  *
00370  * Upon successful return, *kcstring_len contains the length of the
00371  * encoded string.
00372  *
00373  * ASSUMES      Old and new key are always equal to each other, although
00374  *              this may be less than the transform type hash output
00375  *              output length (eg, using KeyChange for a DESPriv key when
00376  *              the user also uses SHA1Auth).  This also implies that the
00377  *              hash placed in the second 1/2 of the key change string
00378  *              will be truncated before the XOR'ing when the hash output is 
00379  *              larger than that 1/2 of the key change string.
00380  *
00381  *              *kcstring_len will be returned as exactly twice that same
00382  *              length though the input buffer may be larger.
00383  *
00384  * XXX FIX:     Does not handle varibable length keys.
00385  * XXX FIX:     Does not handle keys larger than the hash algorithm used.
00386  */
00387 int
00388 encode_keychange(const oid * hashtype, u_int hashtype_len,
00389                  u_char * oldkey, size_t oldkey_len,
00390                  u_char * newkey, size_t newkey_len,
00391                  u_char * kcstring, size_t * kcstring_len)
00392 #if defined(NETSNMP_USE_OPENSSL) || defined(NETSNMP_USE_INTERNAL_MD5) || defined(NETSNMP_USE_PKCS11)
00393 {
00394     int             rval = SNMPERR_SUCCESS;
00395     size_t          properlength;
00396     size_t          nbytes = 0;
00397 
00398     u_char         *tmpbuf = NULL;
00399 
00400 
00401     /*
00402      * Sanity check.
00403      */
00404     if (!kcstring || !kcstring_len)
00405         return SNMPERR_GENERR;
00406 
00407     if (!hashtype || !oldkey || !newkey || !kcstring || !kcstring_len
00408         || (oldkey_len <= 0) || (newkey_len <= 0) || (*kcstring_len <= 0)
00409         || (hashtype_len != USM_LENGTH_OID_TRANSFORM)) {
00410         QUITFUN(SNMPERR_GENERR, encode_keychange_quit);
00411     }
00412 
00413     /*
00414      * Setup for the transform type.
00415      */
00416     properlength = sc_get_properlength(hashtype, hashtype_len);
00417     if (properlength == SNMPERR_GENERR)
00418         QUITFUN(SNMPERR_GENERR, encode_keychange_quit);
00419 
00420     if ((oldkey_len != newkey_len) || (*kcstring_len < (2 * oldkey_len))) {
00421         QUITFUN(SNMPERR_GENERR, encode_keychange_quit);
00422     }
00423 
00424     properlength = SNMP_MIN((int) oldkey_len, properlength);
00425 
00426     /*
00427      * Use the old key and some random bytes to encode the new key
00428      * in the KeyChange TC format:
00429      *      . Get random bytes (store in first half of kcstring),
00430      *      . Hash (oldkey | random_bytes) (into second half of kcstring),
00431      *      . XOR hash and newkey (into second half of kcstring).
00432      *
00433      * Getting the wrong number of random bytes is considered an error.
00434      */
00435     nbytes = properlength;
00436 
00437 #if defined(NETSNMP_ENABLE_TESTING_CODE) && defined(RANDOMZEROS)
00438     memset(kcstring, 0, nbytes);
00439     DEBUGMSG(("encode_keychange",
00440               "** Using all zero bits for \"random\" delta of )"
00441               "the keychange string! **\n"));
00442 #else                           /* !NETSNMP_ENABLE_TESTING_CODE */
00443     rval = sc_random(kcstring, &nbytes);
00444     QUITFUN(rval, encode_keychange_quit);
00445     if ((int) nbytes != properlength) {
00446         QUITFUN(SNMPERR_GENERR, encode_keychange_quit);
00447     }
00448 #endif                          /* !NETSNMP_ENABLE_TESTING_CODE */
00449 
00450     tmpbuf = (u_char *) malloc(properlength * 2);
00451     if (tmpbuf) {
00452         memcpy(tmpbuf, oldkey, properlength);
00453         memcpy(tmpbuf + properlength, kcstring, properlength);
00454 
00455         *kcstring_len -= properlength;
00456         rval = sc_hash(hashtype, hashtype_len, tmpbuf, properlength * 2,
00457                        kcstring + properlength, kcstring_len);
00458 
00459         QUITFUN(rval, encode_keychange_quit);
00460 
00461         *kcstring_len = (properlength * 2);
00462 
00463         kcstring += properlength;
00464         nbytes = 0;
00465         while ((int) (nbytes++) < properlength) {
00466             *kcstring++ ^= *newkey++;
00467         }
00468     }
00469 
00470   encode_keychange_quit:
00471     if (rval != SNMPERR_SUCCESS)
00472         memset(kcstring, 0, *kcstring_len);
00473     SNMP_FREE(tmpbuf);
00474 
00475     return rval;
00476 
00477 }                               /* end encode_keychange() */

00482 /*******************************************************************-o-******
00483  * decode_keychange
00484  *
00485  * Parameters:
00486  *      *hashtype       MIB OID of the hash transform to use.
00487  *       hashtype_len   Length of the hash transform MIB OID.
00488  *      *oldkey         Old key that is used to encode the new key.
00489  *       oldkey_len     Length of oldkey in bytes.
00490  *      *kcstring       Encoded KeyString buffer containing the new key.
00491  *       kcstring_len   Length of kcstring in bytes.
00492  *      *newkey         Buffer to hold the extracted new key.
00493  *      *newkey_len     Length of newkey in bytes.
00494  *      
00495  * Returns:
00496  *      SNMPERR_SUCCESS                 Success.
00497  *      SNMPERR_GENERR                  All errors.
00498  *
00499  *
00500  * Decodes a string of bits encoded according to the KeyChange TC described
00501  * in RFC 2274, Section 5.  The new key is extracted from *kcstring with
00502  * the aid of the old key.
00503  *
00504  * Upon successful return, *newkey_len contains the length of the new key.
00505  *
00506  *
00507  * ASSUMES      Old key is exactly 1/2 the length of the KeyChange buffer,
00508  *              although this length may be less than the hash transform
00509  *              output.  Thus the new key length will be equal to the old
00510  *              key length.
00511  */
00512 /*
00513  * XXX:  if the newkey is not long enough, it should be freed and remalloced 
00514  */
00515 int
00516 decode_keychange(const oid * hashtype, u_int hashtype_len,
00517                  u_char * oldkey, size_t oldkey_len,
00518                  u_char * kcstring, size_t kcstring_len,
00519                  u_char * newkey, size_t * newkey_len)
00520 #if defined(NETSNMP_USE_OPENSSL) || defined(NETSNMP_USE_INTERNAL_MD5) || defined(NETSNMP_USE_PKCS11)
00521 {
00522     int             rval = SNMPERR_SUCCESS;
00523     size_t          properlength = 0;
00524     int             iproperlength = 0;
00525     u_int           nbytes = 0;
00526 
00527     u_char         *bufp, tmp_buf[SNMP_MAXBUF];
00528     size_t          tmp_buf_len = SNMP_MAXBUF;
00529     u_char         *tmpbuf = NULL;
00530 
00531 
00532 
00533     /*
00534      * Sanity check.
00535      */
00536     if (!hashtype || !oldkey || !kcstring || !newkey || !newkey_len
00537         || (oldkey_len <= 0) || (kcstring_len <= 0) || (*newkey_len <= 0)
00538         || (hashtype_len != USM_LENGTH_OID_TRANSFORM)) {
00539         QUITFUN(SNMPERR_GENERR, decode_keychange_quit);
00540     }
00541 
00542 
00543     /*
00544      * Setup for the transform type.
00545      */
00546     iproperlength = sc_get_properlength(hashtype, hashtype_len);
00547     if (iproperlength == SNMPERR_GENERR)
00548         QUITFUN(SNMPERR_GENERR, decode_keychange_quit);
00549 
00550     properlength = (size_t) iproperlength;
00551 
00552     if (((oldkey_len * 2) != kcstring_len) || (*newkey_len < oldkey_len)) {
00553         QUITFUN(SNMPERR_GENERR, decode_keychange_quit);
00554     }
00555 
00556     properlength = oldkey_len;
00557     *newkey_len = properlength;
00558 
00559     /*
00560      * Use the old key and the given KeyChange TC string to recover
00561      * the new key:
00562      *      . Hash (oldkey | random_bytes) (into newkey),
00563      *      . XOR hash and encoded (second) half of kcstring (into newkey).
00564      */
00565     tmpbuf = (u_char *) malloc(properlength * 2);
00566     if (tmpbuf) {
00567         memcpy(tmpbuf, oldkey, properlength);
00568         memcpy(tmpbuf + properlength, kcstring, properlength);
00569 
00570         rval = sc_hash(hashtype, hashtype_len, tmpbuf, properlength * 2,
00571                        tmp_buf, &tmp_buf_len);
00572         QUITFUN(rval, decode_keychange_quit);
00573 
00574         memcpy(newkey, tmp_buf, properlength);
00575         bufp = kcstring + properlength;
00576         nbytes = 0;
00577         while ((int) (nbytes++) < properlength) {
00578             *newkey++ ^= *bufp++;
00579         }
00580     }
00581 
00582   decode_keychange_quit:
00583     if (rval != SNMPERR_SUCCESS) {
00584         memset(newkey, 0, properlength);
00585     }
00586     memset(tmp_buf, 0, SNMP_MAXBUF);
00587     if (tmpbuf != NULL)
00588         SNMP_FREE(tmpbuf);
00589 
00590     return rval;
00591 
00592 }                               /* end decode_keychange() */
