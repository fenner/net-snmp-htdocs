00061 netsnmp_scalar_group_helper_handler(netsnmp_mib_handler *handler,
00062                                 netsnmp_handler_registration *reginfo,
00063                                 netsnmp_agent_request_info *reqinfo,
00064                                 netsnmp_request_info *requests)
00065 {

00072 
00073     DEBUGMSGTL(("helper:scalar_group", "Got request:\n"));
00074     namelen = SNMP_MIN(requests->requestvb->name_length,
00075                        reginfo->rootoid_len);
00076     cmp = snmp_oid_compare(requests->requestvb->name, namelen,
00077                            reginfo->rootoid, reginfo->rootoid_len);
00078 
00079     DEBUGMSGTL(( "helper:scalar_group", "  cmp=%d, oid:", cmp));
00080     DEBUGMSGOID(("helper:scalar_group", var->name, var->name_length));
00081     DEBUGMSG((   "helper:scalar_group", "\n"));
00082 
00083     /*
00084      * copy root oid to root_tmp, set instance to 0. (subid set later on)
00085      * save rootoid, since we'll replace it before calling next handler,
00086      * and need to restore it afterwards.
00087      */
00088     memcpy(root_tmp, reginfo->rootoid, reginfo->rootoid_len * sizeof(oid));
00089     root_tmp[reginfo->rootoid_len + 1] = 0;
00090     root_save = reginfo->rootoid;
00091 
00092     ret = SNMP_ERR_NOCREATION;
00093     switch (reqinfo->mode) {
00094     /*
00095      * The handling of "exact" requests is basically the same.
00096      * The only difference between GET and SET requests is the
00097      *     error/exception to return on failure.
00098      */
00099     case MODE_GET:
00100         ret = SNMP_NOSUCHOBJECT;
00101         /* Fallthrough */
00102 
00103     case MODE_SET_RESERVE1:
00104     case MODE_SET_RESERVE2:
00105     case MODE_SET_ACTION:
00106     case MODE_SET_COMMIT:
00107     case MODE_SET_UNDO:
00108     case MODE_SET_FREE:
00109         if (cmp != 0 ||
00110             requests->requestvb->name_length <= reginfo->rootoid_len) {
00111             /*
00112              * Common prefix doesn't match, or only *just* matches 
00113              *  the registered root (so can't possibly match a scalar)
00114              */
00115             netsnmp_set_request_error(reqinfo, requests, ret);
00116             return SNMP_ERR_NOERROR;
00117         } else {
00118             /*
00119              * Otherwise,
00120              *     extract the object subidentifier from the request, 
00121              *     check this is (probably) valid, and then fudge the
00122              *     registered 'rootoid' to match, before passing the
00123              *     request off to the next handler ('scalar').
00124              *
00125              * Note that we don't bother checking instance subidentifiers
00126              *     here.  That's left to the scalar helper.
00127              */
00128             subid = requests->requestvb->name[reginfo->rootoid_len];
00129             if (subid < sgroup->lbound ||
00130                 subid > sgroup->ubound) {
00131                 netsnmp_set_request_error(reqinfo, requests, ret);
00132                 return SNMP_ERR_NOERROR;
00133             }
00134             root_tmp[reginfo->rootoid_len++] = subid;
00135             reginfo->rootoid = root_tmp;
00136             ret = netsnmp_call_next_handler(handler, reginfo, reqinfo,
00137                                             requests);
00138             reginfo->rootoid = root_save;
00139             reginfo->rootoid_len--;
00140             return ret;
00141         }
00142         break;
00143 
00144     case MODE_GETNEXT:
00145         /*
00146          * If we're being asked for something before (or exactly matches)
00147          *     the registered root OID, then start with the first object.
00148          * If we're being asked for something that exactly matches an object
00149          *    OID, then that's what we pass down.
00150          * Otherwise, we pass down the OID of the *next* object....
00151          */
00152         if (cmp < 0 ||
00153             requests->requestvb->name_length <= reginfo->rootoid_len) {
00154             subid  = sgroup->lbound;
00155         } else if (requests->requestvb->name_length == reginfo->rootoid_len+1)
00156             subid = requests->requestvb->name[reginfo->rootoid_len];
00157         else
00158             subid = requests->requestvb->name[reginfo->rootoid_len]+1;
00159 
00160         /*
00161          * ... always assuming this is (potentially) valid, of course.
00162          */
00163         if (subid > sgroup->ubound)
00164             return SNMP_ERR_NOERROR;
00165         
00166         root_tmp[reginfo->rootoid_len++] = subid;
00167         reginfo->rootoid = root_tmp;
00168         ret = netsnmp_call_next_handler(handler, reginfo, reqinfo,
00169                                             requests);
00170         /*
00171          * If we didn't get an answer (due to holes in the group)
00172          *   set things up to retry again.
00173          */
00174         if (requests->requestvb->type == ASN_NULL ||
00175             requests->requestvb->type == SNMP_NOSUCHOBJECT ||
00176             requests->requestvb->type == SNMP_NOSUCHINSTANCE) {
00177             snmp_set_var_objid(requests->requestvb,
00178                                reginfo->rootoid, reginfo->rootoid_len);
00179             requests->requestvb->name[reginfo->rootoid_len-1] = ++subid;
00180             requests->requestvb->type = ASN_PRIV_RETRY;
00181         }
00182         reginfo->rootoid = root_save;
00183         reginfo->rootoid_len--;
00184         return ret;
00185     }
00186     /*
00187      * got here only if illegal mode found 
00188      */
00189     return SNMP_ERR_GENERR;
00190 }
