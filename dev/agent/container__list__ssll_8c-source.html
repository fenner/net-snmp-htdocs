00057 static void *
00058 _get(netsnmp_container *c, const void *key, int exact)
00059 {
00060     sl_container *sl = (sl_container*)c;
00061     sl_node  *curr = sl->head;
00062     int rc = 0;
00063     
00064     /*
00065      * note: get-next on unsorted list is meaningless. we
00066      * don't try to search whole array, looking for the next highest.
00067      */
00068     if( (NULL != curr) && (NULL != key)) {
00069         while (curr) {
00070             rc = sl->c.compare(curr->data, key);
00071             if (rc == 0)
00072                 break;
00073             else if (rc > 0) {
00074                 if (0 == sl->unsorted) {
00075                     /*
00076                      * if sorted, we can stop.
00077                      */
00078                     break;
00079                 }
00080             }
00081             curr = curr->next;
00082         }
00083         
00084         if((curr) && (!exact) && (rc == 0)) {
00085             curr = curr->next;
00086         }
00087     }
00088
