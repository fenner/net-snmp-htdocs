00076     /*
00077      * this code depends on AUTO_NEXT being set
00078      */
00079     netsnmp_assert(handler->flags & MIB_HANDLER_AUTO_NEXT);
00080 
00081     /*
00082      * don't do anything for any modes besides GETBULK. Just return, and
00083      * the agent will call the next handler (AUTO_NEXT).
00084      *
00085      * for GETBULK, we munge the mode, call the next handler ourselves
00086      * (setting AUTO_NEXT_OVERRRIDE so the agent knows what we did),
00087      * restore the mode and fix up the requests.
00088      */
00089     if(MODE_GETBULK == reqinfo->mode) {
00090         reqinfo->mode = MODE_GETNEXT;
00091         ret =
00092             netsnmp_call_next_handler(handler, reginfo, reqinfo, requests);
00093         reqinfo->mode = MODE_GETBULK;
00094 
00095         /*
00096          * update the varbinds for the next request series 
00097          */
00098         netsnmp_bulk_to_next_fix_requests(requests);
00099 
00100         /*
00101          * let agent handler know that we've already called next handler
00102          */
00103         handler->flags |= MIB_HANDLER_AUTO_NEXT_OVERRIDE_ONCE;
00104     }
00105 
00106     return ret;
00107 }
